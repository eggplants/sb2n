 22def _count_blocks_recursive(block) -> int:
 23    """Count blocks recursively including nested children.
 24
 25    Args:
 26        block: A Notion block object
 27
 28    Returns:
 29        Total count of blocks including nested children
 30    """
 31    count = 1  # Count the block itself
 32    # Check for children in bulleted_list_item
 33    if hasattr(block, "bulleted_list_item") and isinstance(block.bulleted_list_item, dict):
 34        children = block.bulleted_list_item.get("children", [])
 35        for child in children:
 36            count += _count_blocks_recursive(child)
 37    return count

 40def _split_blocks_into_chunks(blocks: list, max_blocks: int = 1000) -> list[list]:
 41    """Split blocks into chunks that don't exceed max_blocks (including nested children).
 42
 43    Args:
 44        blocks: List of top-level blocks
 45        max_blocks: Maximum number of blocks per chunk (including nested children)
 46
 47    Returns:
 48        List of block chunks
 49    """
 50    chunks = []
 51    current_chunk = []
 52    current_count = 0
 53
 54    for block in blocks:
 55        block_count = _count_blocks_recursive(block)
 56
 57        # If adding this block would exceed the limit, start a new chunk
 58        if current_chunk and current_count + block_count > max_blocks:
 59            chunks.append(current_chunk)
 60            current_chunk = []
 61            current_count = 0
 62
 63        current_chunk.append(block)
 64        current_count += block_count
 65
 66    # Add the last chunk if it has any blocks
 67    if current_chunk:
 68        chunks.append(current_chunk)
 69
 70    return chunks

153    def migrate_all(self) -> MigrationSummary:
154        """Migrate all pages from Scrapbox to Notion.
155
156        Returns:
157            Summary of migration results
158        """
159        logger.info("Starting migration from Scrapbox to Notion")
160        if self.dry_run:
161            logger.info("DRY RUN MODE: No actual changes will be made")
162
163        # Validate database properties before migration
164        if not self.dry_run:
165            try:
166                self.notion_service.validate_database_properties()
167            except Exception:
168                logger.error("Database validation failed. Migration aborted.")
169                raise
170
171        # Get existing pages if skip_existing is enabled
172        existing_titles: set[str] = set()
173        if self.skip_existing:
174            try:
175                existing_titles = self.notion_service.get_existing_page_titles()
176            except Exception:
177                logger.exception("Failed to get existing pages")
178                logger.info("Continuing without skip-existing functionality")
179
180        results: list[MigrationResult] = []
181
182        with ScrapboxService(self.config.scrapbox_project, self.config.scrapbox_connect_sid) as scrapbox:
183            # Initialize converter with scrapbox service for image downloads
184            self.converter = NotionBlockConverter(self.notion_service, scrapbox, enable_icon=self.enable_icon)
185
186            # If specific pages are requested, process them directly
187            if self.page_titles:
188                logger.info(
189                    "Processing %(count)d specific pages by --pages option",
190                    {"count": len(self.page_titles)},
191                )
192                pages_to_process = self.page_titles
193                # Apply limit if specified
194                if self.limit is not None:
195                    pages_to_process = pages_to_process[: self.limit]
196                    logger.info(
197                        "Limiting to %(migrating)d pages by -n option",
198                        {"migrating": len(pages_to_process)},
199                    )
200            else:
201                # Get all pages
202                all_pages = scrapbox.get_all_pages()
203
204                # Apply limit if specified
205                if self.limit is not None:
206                    pages_list = all_pages[: self.limit]
207                    logger.info(
208                        "Found %(total)d pages, migrating %(migrating)d pages (limited by -n option)",
209                        {"total": len(all_pages), "migrating": len(pages_list)},
210                    )
211                else:
212                    pages_list = all_pages
213                    logger.info("Found %(count)d pages to migrate", {"count": len(pages_list)})
214
215                pages_to_process = [p.title for p in pages_list]
216
217            total = len(pages_to_process)
218
219            # Migrate each page
220            for i, page_title in enumerate(pages_to_process, 1):
221                logger.info(
222                    "Processing page %(current)d/%(total)d: %(title)s",
223                    {"current": i, "total": total, "title": page_title},
224                )
225
226                # Check if page already exists and should be skipped
227                if self.skip_existing and page_title in existing_titles:
228                    logger.info(" Skipping existing page: %(title)s", {"title": page_title})
229                    results.append(
230                        MigrationResult(
231                            page_title=page_title,
232                            success=True,
233                            notion_page_id=SpecialPageId.SKIPPED_ID,
234                        )
235                    )
236                    continue
237
238                try:
239                    result = self._migrate_page(scrapbox, page_title, existing_titles)
240                    results.append(result)
241
242                    if result.success:
243                        logger.info(" Successfully migrated: %(title)s", {"title": page_title})
244                    else:
245                        logger.error(
246                            " Failed to migrate: %(title)s - %(error)s", {"title": page_title, "error": result.error}
247                        )
248
249                except Exception as e:
250                    logger.exception("Unexpected error migrating %(title)s", {"title": page_title})
251                    results.append(
252                        MigrationResult(
253                            page_title=page_title,
254                            success=False,
255                            error=str(e),
256                        )
257                    )
258
259        # Calculate summary
260        skipped_count = sum(1 for r in results if r.success and r.notion_page_id == SpecialPageId.SKIPPED_ID)
261        successful_count = sum(1 for r in results if r.success and r.notion_page_id != SpecialPageId.SKIPPED_ID)
262        summary = MigrationSummary(
263            total_pages=total,
264            successful=successful_count,
265            failed=sum(1 for r in results if not r.success),
266            skipped=skipped_count,
267            results=results,
268        )
269
270        self._print_summary(summary)
271        return summary

273    def _migrate_page(
274        self,
275        scrapbox: ScrapboxService,
276        page_title: str,
277        existing_titles: set[str],
278    ) -> MigrationResult:
279        """Migrate a single page.
280
281        Args:
282            scrapbox: Scrapbox service
283            page_title: Title of the page to migrate
284            existing_titles: Set of existing page titles in Notion (for skip-existing check)
285
286        Returns:
287            Migration result for this page
288        """
289        try:
290            # Get page text
291            page_text = scrapbox.get_page_text(page_title)
292
293            page_detail = scrapbox.get_page_detail(page_title)
294            created_timestamp = page_detail.created
295
296            # Extract tags
297            tags = [tag.replace(",", "") for tag in ScrapboxParser.extract_tags(page_text)]
298
299            # Convert creation date
300            created_date = datetime.fromtimestamp(created_timestamp, tz=UTC)
301
302            # Generate Scrapbox URL
303            scrapbox_url = scrapbox.get_page_url(page_title)
304
305            if self.dry_run:
306                logger.debug("[DRY RUN] Would create page: %(title)s", {"title": page_title})
307                logger.debug("  Tags: %(tags)s", {"tags": tags})
308                logger.debug("  Created: %(created)s", {"created": created_date})
309                logger.debug("  URL: %(url)s", {"url": scrapbox_url})
310                return MigrationResult(
311                    page_title=page_title,
312                    success=True,
313                    notion_page_id=SpecialPageId.DRY_RUN_ID,
314                )
315
316            # Convert blocks first to determine if splitting is needed
317            blocks = []
318            total_block_count = 0
319            chunks = []
320            total_chunks = 1
321
322            if self.converter:
323                blocks = self.converter.convert_to_blocks(page_text)
324                total_block_count = sum(_count_blocks_recursive(block) for block in blocks)
325                max_block_count = 1000
326
327                # Check if we need to split into multiple pages
328                if total_block_count > max_block_count:
329                    chunks = _split_blocks_into_chunks(blocks, max_block_count)
330                    total_chunks = len(chunks)
331
332                    logger.warning(
333                        "Page '%(title)s' has %(total)d blocks (including nested children), "
334                        "which exceeds the limit of %(max)d. This page will be split into %(chunks)d pages.",
335                        {
336                            "title": page_title,
337                            "total": total_block_count,
338                            "max": max_block_count,
339                            "chunks": total_chunks,
340                        },
341                    )
342
343            # Check if split pages already exist (for skip-existing)
344            if self.skip_existing and total_chunks > 1:
345                all_split_pages_exist = True
346                for i in range(1, total_chunks + 1):
347                    split_title = f"{page_title} - {i}/{total_chunks}"
348                    if split_title not in existing_titles:
349                        all_split_pages_exist = False
350                        break
351
352                if all_split_pages_exist:
353                    logger.info(
354                        " Skipping page '%(title)s' - all %(chunks)d split pages already exist",
355                        {"title": page_title, "chunks": total_chunks},
356                    )
357                    return MigrationResult(
358                        page_title=page_title,
359                        success=True,
360                        notion_page_id=SpecialPageId.SKIPPED_ID,
361                    )
362
363            # Determine the title for the first page
364            first_page_title = f"{page_title} - 1/{total_chunks}" if total_chunks > 1 else page_title
365
366            # Check if the first page already exists (for skip-existing with split pages)
367            if self.skip_existing and total_chunks > 1 and first_page_title in existing_titles:
368                logger.info(
369                    " Skipping first split page: '%(title)s' (already exists)",
370                    {"title": first_page_title},
371                )
372                # Skip creating the first page, but continue to check other pages
373                notion_page_id = None
374            else:
375                # Create Notion page with appropriate title
376                notion_page = self.notion_service.create_database_page(
377                    title=first_page_title,
378                    scrapbox_url=scrapbox_url,
379                    created_date=created_date,
380                    tags=tags,
381                )
382                notion_page_id = notion_page["id"]
383
384            # Append blocks
385            if self.converter and blocks:
386                try:
387                    # Log block statistics
388                    block_types = {}
389                    for block in blocks:
390                        block_type = block.type
391                        block_types[block_type] = block_types.get(block_type, 0) + 1
392
393                    logger.debug(
394                        "Block statistics for page '%(title)s': %(stats)s (total: %(total)d including nested)",
395                        {"title": page_title, "stats": block_types, "total": total_block_count},
396                    )
397
398                    if total_chunks > 1:
399                        # Append first chunk to the already created page (if not skipped)
400                        if notion_page_id is not None:
401                            first_chunk = chunks[0]
402                            first_chunk_count = sum(_count_blocks_recursive(block) for block in first_chunk)
403                            logger.info(
404                                "Appending chunk 1/%(total)d to page '%(title)s' (%(count)d blocks)",
405                                {"total": total_chunks, "title": first_page_title, "count": first_chunk_count},
406                            )
407                            self.notion_service.append_blocks(notion_page_id, first_chunk)
408
409                        # Create additional pages for remaining chunks
410                        for i, chunk in enumerate(chunks[1:], start=2):
411                            chunk_count = sum(_count_blocks_recursive(block) for block in chunk)
412                            chunk_page_title = f"{page_title} - {i}/{total_chunks}"
413
414                            # Check if this split page already exists (for skip-existing)
415                            if self.skip_existing and chunk_page_title in existing_titles:
416                                logger.info(
417                                    " Skipping split page %(page_num)d/%(total)d: '%(title)s' (already exists)",
418                                    {"page_num": i, "total": total_chunks, "title": chunk_page_title},
419                                )
420                                continue
421
422                            logger.info(
423                                "Creating additional page %(page_num)d/%(total)d: '%(title)s' (%(count)d blocks)",
424                                {"page_num": i, "total": total_chunks, "title": chunk_page_title, "count": chunk_count},
425                            )
426
427                            # Create new page with the same metadata
428                            chunk_page = self.notion_service.create_database_page(
429                                title=chunk_page_title,
430                                scrapbox_url=scrapbox_url,
431                                created_date=created_date,
432                                tags=tags,
433                            )
434                            chunk_page_id = chunk_page["id"]
435
436                            # Append blocks to the new page
437                            self.notion_service.append_blocks(chunk_page_id, chunk)
438
439                    elif notion_page_id is not None:
440                        logger.debug(
441                            "Appending %(count)d blocks to page '%(title)s'",
442                            {"count": len(blocks), "title": page_title},
443                        )
444                        self.notion_service.append_blocks(notion_page_id, blocks)
445
446                except Exception:
447                    # If block append fails, delete the created page
448                    logger.exception(
449                        "Failed to append blocks to page '%(title)s'. Deleting the page...",
450                        {"title": page_title},
451                    )
452                    if notion_page_id is not None:
453                        try:
454                            self.notion_service.delete_page(notion_page_id)
455                            logger.info("Successfully deleted incomplete page: %(title)s", {"title": page_title})
456                        except Exception:
457                            logger.exception(
458                                "Failed to delete incomplete page '%(title)s'",
459                                {"title": page_title},
460                            )
461                    # Re-raise the original error
462                    raise
463
464            return MigrationResult(
465                page_title=page_title,
466                success=True,
467                notion_page_id=notion_page_id,
468            )
469
470        except Exception as e:
471            logger.exception("Error migrating page: %(title)s", {"title": page_title})
472            return MigrationResult(
473                page_title=page_title,
474                success=False,
475                error=str(e),
476            )

276        scrapbox: ScrapboxService,
277        page_title: str,
278        existing_titles: set[str],
279    ) -> MigrationResult:
280        """Migrate a single page.
281
282        Args:
283            scrapbox: Scrapbox service
284            page_title: Title of the page to migrate
285            existing_titles: Set of existing page titles in Notion (for skip-existing check)
286
287        Returns:
288            Migration result for this page

291            # Get page text

293
294            page_detail = scrapbox.get_page_detail(page_title)
295            created_timestamp = page_detail.created
296
297            # Extract tags
298            tags = [tag.replace(",", "") for tag in ScrapboxParser.extract_tags(page_text)]

300            # Convert creation date
301            created_date = datetime.fromtimestamp(created_timestamp, tz=UTC)
302
303            # Generate Scrapbox URL
304            scrapbox_url = scrapbox.get_page_url(page_title)
305
306            if self.dry_run:
307                logger.debug("[DRY RUN] Would create page: %(title)s", {"title": page_title})
308                logger.debug("  Tags: %(tags)s", {"tags": tags})
309                logger.debug("  Created: %(created)s", {"created": created_date})
310                logger.debug("  URL: %(url)s", {"url": scrapbox_url})
311                return MigrationResult(
312                    page_title=page_title,
313                    success=True,
314                    notion_page_id=SpecialPageId.DRY_RUN_ID,
315                )
316
317            # Convert blocks first to determine if splitting is needed
318            blocks = []
319            total_block_count = 0
320            chunks = []
321            total_chunks = 1
322

324                blocks = self.converter.convert_to_blocks(page_text)
325                total_block_count = sum(_count_blocks_recursive(block) for block in blocks)
326                max_block_count = 1000
327
328                # Check if we need to split into multiple pages
329                if total_block_count > max_block_count:
330                    chunks = _split_blocks_into_chunks(blocks, max_block_count)
331                    total_chunks = len(chunks)
332
333                    logger.warning(
334                        "Page '%(title)s' has %(total)d blocks (including nested children), "
335                        "which exceeds the limit of %(max)d. This page will be split into %(chunks)d pages.",

337                            "title": page_title,
338                            "total": total_block_count,
339                            "max": max_block_count,
340                            "chunks": total_chunks,
341                        },
342                    )
343
344            # Check if split pages already exist (for skip-existing)
345            if self.skip_existing and total_chunks > 1:
346                all_split_pages_exist = True
347                for i in range(1, total_chunks + 1):
348                    split_title = f"{page_title} - {i}/{total_chunks}"
349                    if split_title not in existing_titles:
350                        all_split_pages_exist = False
351                        break
352
353                if all_split_pages_exist:
354                    logger.info(
355                        " Skipping page '%(title)s' - all %(chunks)d split pages already exist",
356                        {"title": page_title, "chunks": total_chunks},
357                    )
358                    return MigrationResult(
359                        page_title=page_title,
360                        success=True,
361                        notion_page_id=SpecialPageId.SKIPPED_ID,
362                    )
363
364            # Determine the title for the first page
365            first_page_title = f"{page_title} - 1/{total_chunks}" if total_chunks > 1 else page_title
366
367            # Check if the first page already exists (for skip-existing with split pages)
368            if self.skip_existing and total_chunks > 1 and first_page_title in existing_titles:
369                logger.info(

371                    {"title": first_page_title},
372                )
373                # Skip creating the first page, but continue to check other pages
374                notion_page_id = None
375            else:
376                # Create Notion page with appropriate title
377                notion_page = self.notion_service.create_database_page(
378                    title=first_page_title,
379                    scrapbox_url=scrapbox_url,
380                    created_date=created_date,
381                    tags=tags,
382                )
383                notion_page_id = notion_page["id"]
384
385            # Append blocks
386            if self.converter and blocks:
387                try:
388                    # Log block statistics
389                    block_types = {}
390                    for block in blocks:
391                        block_type = block.type
392                        block_types[block_type] = block_types.get(block_type, 0) + 1
393
394                    logger.debug(
395                        "Block statistics for page '%(title)s': %(stats)s (total: %(total)d including nested)",
396                        {"title": page_title, "stats": block_types, "total": total_block_count},
397                    )
398
399                    if total_chunks > 1:
400                        # Append first chunk to the already created page (if not skipped)
401                        if notion_page_id is not None:
402                            first_chunk = chunks[0]
403                            first_chunk_count = sum(_count_blocks_recursive(block) for block in first_chunk)
404                            logger.info(
405                                "Appending chunk 1/%(total)d to page '%(title)s' (%(count)d blocks)",
406                                {"total": total_chunks, "title": first_page_title, "count": first_chunk_count},
407                            )
408                            self.notion_service.append_blocks(notion_page_id, first_chunk)
409
410                        # Create additional pages for remaining chunks
411                        for i, chunk in enumerate(chunks[1:], start=2):
412                            chunk_count = sum(_count_blocks_recursive(block) for block in chunk)
413                            chunk_page_title = f"{page_title} - {i}/{total_chunks}"
414
415                            # Check if this split page already exists (for skip-existing)
416                            if self.skip_existing and chunk_page_title in existing_titles:
417                                logger.info(
418                                    " Skipping split page %(page_num)d/%(total)d: '%(title)s' (already exists)",
419                                    {"page_num": i, "total": total_chunks, "title": chunk_page_title},
420                                )
421                                continue
422
423                            logger.info(
424                                "Creating additional page %(page_num)d/%(total)d: '%(title)s' (%(count)d blocks)",
425                                {"page_num": i, "total": total_chunks, "title": chunk_page_title, "count": chunk_count},
426                            )
427
428                            # Create new page with the same metadata
429                            chunk_page = self.notion_service.create_database_page(
430                                title=chunk_page_title,
431                                scrapbox_url=scrapbox_url,
432                                created_date=created_date,
433                                tags=tags,
434                            )
435                            chunk_page_id = chunk_page["id"]
436
437                            # Append blocks to the new page
438                            self.notion_service.append_blocks(chunk_page_id, chunk)
439
440                    elif notion_page_id is not None:
441                        logger.debug(
442                            "Appending %(count)d blocks to page '%(title)s'",
443                            {"count": len(blocks), "title": page_title},
444                        )
445                        self.notion_service.append_blocks(notion_page_id, blocks)
446
447                except Exception:
448                    # If block append fails, delete the created page
449                    logger.exception(
450                        "Failed to append blocks to page '%(title)s'. Deleting the page...",
451                        {"title": page_title},
452                    )
453                    if notion_page_id is not None:
454                        try:
455                            self.notion_service.delete_page(notion_page_id)
456                            logger.info("Successfully deleted incomplete page: %(title)s", {"title": page_title})
457                        except Exception:
458                            logger.exception(
459                                "Failed to delete incomplete page '%(title)s'",
460                                {"title": page_title},
461                            )
462                    # Re-raise the original error
463                    raise
464
465            return MigrationResult(
466                page_title=page_title,
467                success=True,
468                notion_page_id=notion_page_id,
469            )
470
471        except Exception as e:
472            logger.exception("Error migrating page: %(title)s", {"title": page_title})
473            return MigrationResult(
474                page_title=page_title,
475                success=False,
476                error=str(e),
477            )
