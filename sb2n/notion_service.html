 75    def get_existing_page_titles(self) -> set[str]:
 76        """Get all existing page titles from the database.
 77
 78        Returns:
 79            Set of page titles currently in the database
 80        """
 81        logger.info("Fetching existing pages from Notion database")
 82        existing_titles: set[str] = set()
 83
 84        try:
 85            # Query the database with pagination
 86            has_more = True
 87            start_cursor = None
 88
 89            while has_more:
 90                # Create query request
 91                query_request = QueryDatabaseRequest(
 92                    database_id=UUID(self.database_id),
 93                    page_size=100,
 94                    start_cursor=start_cursor,
 95                )
 96
 97                # Execute query
 98                response_dict = self.client.databases.query(**query_request.model_dump(mode="json", exclude_none=True))  # ty:ignore[possibly-missing-attribute]
 99                response = QueryDatabaseResponse.model_validate(response_dict)
100
101                for page in response.results:
102                    # Extract title from properties
103                    properties = page.properties if hasattr(page, "properties") else {}
104                    title_prop = properties.get("Title") or properties.get("Name")  # ty:ignore[unresolved-attribute]
105                    if title_prop and hasattr(title_prop, "title"):
106                        title_content = title_prop.title
107                        if title_content:
108                            title_text = title_content[0].plain_text if hasattr(title_content[0], "plain_text") else ""
109                            if title_text:
110                                existing_titles.add(title_text)
111
112                has_more = response.has_more or False
113                start_cursor = response.next_cursor
114
115            logger.info("Found %(count)d existing pages in Notion", {"count": len(existing_titles)})
116        except Exception:
117            logger.exception("Failed to fetch existing pages from Notion")
118            raise
119        else:
120            return existing_titles

306    def upload_image(self, image_data: bytes, filename: str = "image.png") -> str:
307        """Upload an image to Notion using file_uploads API.
308
309        Args:
310            image_data: Binary image data
311            filename: Filename for the image (optional)
312
313        Returns:
314            File upload ID to use in blocks
315
316        Raises:
317            Exception: If upload fails
318        """
319        try:
320            # Step 1: Create file upload
321            raw_response = self.client.file_uploads.create(mode="single_part")
322            logger.debug("Raw file upload response: %s", raw_response)
323            file_upload = FileUploadResponse.model_validate(raw_response)
324            logger.debug("Created file upload with ID: %(file_upload_id)s", {"file_upload_id": file_upload.id})
325
326            # Step 2: Send file data
327            file_obj = BytesIO(image_data)
328            file_obj.name = filename
329            self.client.file_uploads.send(
330                file_upload_id=file_upload.id,
331                file=file_obj,
332            )
333            logger.debug("Uploaded image to Notion: %(filename)s", {"filename": filename})
334        except Exception:
335            logger.exception("Failed to upload image: %(filename)s", {"filename": filename})
336            raise
337        else:
338            return file_upload.id

368    def _convert_rich_text_elements(self, elements: list[RichTextElement]) -> list[dict]:
369        """Convert RichTextElement list to Notion rich_text format.
370
371        Args:
372            elements: List of rich text elements
373
374        Returns:
375            List of Notion rich_text objects
376        """
377        result = []
378        for elem in elements:
379            annotations = {
380                "bold": elem.bold,
381                "italic": elem.italic,
382                "strikethrough": elem.strikethrough,
383                "underline": elem.underline,
384                "code": elem.code,
385                "color": "default",
386            }
387
388            if elem.link_url:
389                # Link with annotations
390                result.append(
391                    {
392                        "type": "text",
393                        "text": {"content": elem.text, "link": {"url": elem.link_url}},
394                        "annotations": annotations,
395                    }
396                )
397            else:
398                # Plain text with annotations
399                result.append(
400                    {
401                        "type": "text",
402                        "text": {"content": elem.text},
403                        "annotations": annotations,
404                    }
405                )
406
407        return result
