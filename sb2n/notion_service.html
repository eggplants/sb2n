 75    def get_existing_page_titles(self) -> set[str]:
 76        """Get all existing page titles from the database.
 77
 78        Returns:
 79            Set of page titles currently in the database
 80        """
 81        logger.info("Fetching existing pages from Notion database")
 82        existing_titles: set[str] = set()
 83
 84        try:
 85            # Step 1: Retrieve database to get data sources
 86            database = self.client.databases.retrieve(database_id=self.database_id)
 87            data_sources = database.get("data_sources", [])  # ty:ignore[possibly-missing-attribute]
 88
 89            if not data_sources:
 90                logger.warning("No data sources found in database")
 91                return existing_titles
 92
 93            # Step 2: Query each data source for pages
 94            for data_source in data_sources:
 95                data_source_id = data_source.get("id")
 96                if not data_source_id:
 97                    continue
 98
 99                logger.debug("Querying data source: %(id)s", {"id": data_source_id})
100
101                # Query data source with pagination
102                has_more = True
103                start_cursor = None
104
105                while has_more:
106                    # Build request body for data_sources query
107                    body: dict[str, Any] = {"page_size": 100}
108                    if start_cursor:
109                        body["start_cursor"] = start_cursor
110
111                    # Execute query using data_sources API
112                    response_dict = self.client.request(
113                        method="POST",
114                        path=f"data_sources/{data_source_id}/query",
115                        body=body,
116                    )
117                    response = QueryDatabaseResponse.model_validate(response_dict)
118
119                    for page in response.results:
120                        # Extract title from properties
121                        properties = page.get("properties", {})
122                        title_prop = properties.get("Title") or properties.get("Name")
123                        if title_prop and title_prop.get("type") == "title":  # noqa: PLR2004
124                            title_content = title_prop.get("title", [])
125                            if title_content:
126                                title_text = title_content[0].get("plain_text", "")
127                                if title_text:
128                                    existing_titles.add(title_text)
129
130                    has_more = response.has_more
131                    start_cursor = response.next_cursor
132
133            logger.info("Found %(count)d existing pages in Notion", {"count": len(existing_titles)})
134        except Exception:
135            logger.exception("Failed to fetch existing pages from Notion")
136            raise
137        else:
138            return existing_titles

141        self,
142        title: str,
143        scrapbox_url: str,
144        created_date: datetime,
145        tags: list[str],
146    ) -> dict | SyncAsync:
147        """Create a new page in the Notion database.
148
149        Args:
150            title: Page title

152            created_date: Original creation date from Scrapbox

157
158        Raises:

160        """

216    def append_blocks(self, page_id: UUID, blocks: list[BlockObject]) -> None:
217        """Append blocks to a Notion page.
218
219        Args:
220            page_id: Notion page ID
221            blocks: List of block objects to append (can include table blocks with children)
222
223        Raises:
224            Exception: If block append fails
225        """
226        if not blocks:
227            logger.debug("No blocks to append for page: %(page_id)s", {"page_id": page_id})
228            return
229
230        logger.debug("Appending %(count)d blocks to page: %(page_id)s", {"count": len(blocks), "page_id": page_id})
231
232        try:
233            # Notion API has a limit of 100 blocks per request
234            batch_size = 100
235            for i in range(0, len(blocks), batch_size):
236                batch = blocks[i : i + batch_size]
237                # Convert pydantic models to dicts, or use dict directly
238                batch_dicts = []
239                for block in batch:
240                    # Handle table blocks with children
241                    if isinstance(block, TableBlockWithChildren):
242                        # Convert table block and its children
243                        table_dict = block.block.model_dump(mode="json", exclude_none=True)
244                        # Add children to the table object, not at the top level
245                        table_dict["table"]["children"] = [
246                            child.model_dump(mode="json", exclude_none=True) for child in block.children
247                        ]
248                        batch_dicts.append(table_dict)
249                    elif hasattr(block, "model_dump"):
250                        batch_dicts.append(block.model_dump(mode="json", exclude_none=True))
251                    else:
252                        batch_dicts.append(block)
253
254                self.client.blocks.children.append(block_id=str(page_id), children=batch_dicts)
255                logger.debug(
256                    "Appended batch %(batch_num)d (%(count)d blocks)",
257                    {"batch_num": i // batch_size + 1, "count": len(batch)},
258                )
259
260            logger.info("Successfully appended %(count)d blocks", {"count": len(blocks)})
261        except Exception:
262            logger.exception("Failed to append blocks to page: %(page_id)s", {"page_id": page_id})
263            raise

423    def create_table_block(
424        self, table_rows: list[list[str]], *, has_column_header: bool = True, has_row_header: bool = False
425    ) -> TableBlockWithChildren:
426        """Create a table block with rows.
427
428        Args:
429            table_rows: List of rows, each row is a list of cell contents
430            has_column_header: Whether the first row is a header
431            has_row_header: Whether the first column is a header
432
433        Returns:
434            TableBlockWithChildren instance
435        """
436        if not table_rows:
437            msg = "Table must have at least one row"
438            raise ValueError(msg)
439
440        # Determine table width from the maximum row length
441        table_width = max(len(row) for row in table_rows)
442
443        # Normalize all rows to have the same width (pad with empty strings)
444        normalized_rows = [
445            row + [""] * (table_width - len(row)) if len(row) < table_width else row for row in table_rows
446        ]
447        # Create table block
448        table_block = TableBlock.new(
449            table_width=table_width, has_column_header=has_column_header, has_row_header=has_row_header
450        )
451
452        # Create table row blocks
453        row_blocks = [TableRowBlock.new(cells=row) for row in normalized_rows]
454
455        return TableBlockWithChildren(block=table_block, children=row_blocks)

457    def _convert_rich_text_elements(self, elements: list[RichTextElement]) -> list[dict]:
458        """Convert RichTextElement list to Notion rich_text format.
459
460        Args:
461            elements: List of rich text elements
462
463        Returns:
464            List of Notion rich_text objects
465        """
466        result = []
467        for elem in elements:
468            annotations = {
469                "bold": elem.bold,
470                "italic": elem.italic,
471                "strikethrough": elem.strikethrough,
472                "underline": elem.underline,
473                "code": elem.code,
474                "color": elem.background_color if elem.background_color else "default",
475            }
476
477            if elem.link_url:
478                # Link with annotations
479                result.append(
480                    {
481                        "type": "text",
482                        "text": {"content": elem.text, "link": {"url": elem.link_url}},
483                        "annotations": annotations,
484                    }
485                )
486            else:
487                # Plain text with annotations
488                result.append(
489                    {
490                        "type": "text",
491                        "text": {"content": elem.text},
492                        "annotations": annotations,
493                    }
494                )
495
496        return result

560        """Get all blocks from a page, optionally including child blocks.
561
562        Args:

565
566        Returns:
567            List of block objects
568        """
569        all_blocks: list[dict[str, Any]] = []
570

573            has_more = True
574            start_cursor = None
575

580
581                response = self.client.blocks.children.list(**params)
582                blocks = response.get("results", [])  # ty:ignore[possibly-missing-attribute]
583                all_blocks.extend(blocks)
584
585                has_more = response.get("has_more", False)  # ty:ignore[possibly-missing-attribute]
586                start_cursor = response.get("next_cursor")  # ty:ignore[possibly-missing-attribute]
587
588            # Recursively fetch child blocks if requested
589            if recursive:
590                for block in all_blocks[:]:  # Copy list to avoid modification during iteration
591                    if block.get("has_children"):
592                        child_blocks = self.get_page_blocks(block["id"], recursive=True)
593                        all_blocks.extend(child_blocks)
594
595        except Exception:
596            logger.exception("Failed to fetch blocks for page: %(page_id)s", {"page_id": page_id})
597            raise

225            Exception: If block append fails
226        """

229            return

231        logger.debug("Appending %(count)d blocks to page: %(page_id)s", {"count": len(blocks), "page_id": page_id})
232
233        try:
234            # Notion API has a limit of 100 blocks per request
235            batch_size = 100
236            for i in range(0, len(blocks), batch_size):
237                batch = blocks[i : i + batch_size]
238                # Convert pydantic models to dicts, or use dict directly
239                batch_dicts = []
240                for block in batch:
241                    # Handle table blocks with children
242                    if isinstance(block, TableBlockWithChildren):
243                        # Convert table block and its children
244                        table_dict = block.block.model_dump(mode="json", exclude_none=True)
245                        # Add children to the table object, not at the top level
246                        table_dict["table"]["children"] = [
247                            child.model_dump(mode="json", exclude_none=True) for child in block.children
248                        ]
249                        batch_dicts.append(table_dict)
250                    elif hasattr(block, "model_dump"):
251                        batch_dicts.append(block.model_dump(mode="json", exclude_none=True))
252                    else:
253                        batch_dicts.append(block)
254
255                self.client.blocks.children.append(block_id=str(page_id), children=batch_dicts)
256                logger.debug(
257                    "Appended batch %(batch_num)d (%(count)d blocks)",
258                    {"batch_num": i // batch_size + 1, "count": len(batch)},

424    def create_table_block(
425        self, table_rows: list[list[str]], *, has_column_header: bool = True, has_row_header: bool = False
426    ) -> TableBlockWithChildren:
427        """Create a table block with rows.
428
429        Args:
430            table_rows: List of rows, each row is a list of cell contents
431            has_column_header: Whether the first row is a header
432            has_row_header: Whether the first column is a header
433
434        Returns:
435            TableBlockWithChildren instance
436        """
437        if not table_rows:
438            msg = "Table must have at least one row"
439            raise ValueError(msg)
440
441        # Determine table width from the maximum row length
442        table_width = max(len(row) for row in table_rows)
443
444        # Normalize all rows to have the same width (pad with empty strings)
445        normalized_rows = [
446            row + [""] * (table_width - len(row)) if len(row) < table_width else row for row in table_rows
447        ]
448        # Create table block
449        table_block = TableBlock.new(
450            table_width=table_width, has_column_header=has_column_header, has_row_header=has_row_header
451        )
452
453        # Create table row blocks
454        row_blocks = [TableRowBlock.new(cells=row) for row in normalized_rows]
455
456        return TableBlockWithChildren(block=table_block, children=row_blocks)

488                # Plain text with annotations

491                        "type": "text",
492                        "text": {"content": elem.text},
