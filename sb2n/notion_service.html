 80    def validate_database_properties(self) -> None:
 81        """Validate that required database properties exist.
 82
 83        Checks that the database has the following required properties:
 84        - Title (or Name) - title type
 85        - Scrapbox URL - url type
 86        - Created Date - date type
 87
 88        Optional properties:
 89        - Tags - multi_select type
 90
 91        Raises:
 92            DatabasePropertyValidationError: If required properties are missing or have incorrect types
 93        """
 94        logger.info("Validating Notion database properties")
 95
 96        try:
 97            # Retrieve database to get data sources (Notion API v2025-09-03)
 98            database = self.client.databases.retrieve(database_id=self.database_id)
 99            data_sources = database.get("data_sources", [])  # ty:ignore[unresolved-attribute]
100
101            if not data_sources:
102                msg = "Database has no data sources"
103                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
104
105            # Retrieve the first data source to get properties
106            data_source_id = data_sources[0].get("id")
107            if not data_source_id:
108                msg = "Data source has no ID"
109                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
110
111            data_source = self.client.data_sources.retrieve(data_source_id=data_source_id)
112            properties = data_source.get("properties", {})  # ty:ignore[unresolved-attribute]
113
114            if not properties:
115                msg = "Database has no properties"
116                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
117
118            # Check for title property (Title or Name)
119            title_prop = properties.get("Title") or properties.get("Name")
120            if not title_prop:
121                msg = "Database must have a 'Title' or 'Name' property"
122                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
123            if title_prop.get("type") != "title":  # noqa: PLR2004
124                msg = f"'Title' or 'Name' property must be of type 'title', got '{title_prop.get('type')}'"
125                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
126
127            # Check for Scrapbox URL property
128            url_prop = properties.get("Scrapbox URL")
129            if not url_prop:
130                msg = "Database must have a 'Scrapbox URL' property"
131                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
132            if url_prop.get("type") != "url":  # noqa: PLR2004
133                msg = f"'Scrapbox URL' property must be of type 'url', got '{url_prop.get('type')}'"
134                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
135
136            # Check for Created Date property
137            date_prop = properties.get("Created Date")
138            if not date_prop:
139                msg = "Database must have a 'Created Date' property"
140                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
141            if date_prop.get("type") != "date":  # noqa: PLR2004
142                msg = f"'Created Date' property must be of type 'date', got '{date_prop.get('type')}'"
143                raise DatabasePropertyValidationError(msg)  # noqa: TRY301
144
145            # Check for optional Tags property (warn if missing or wrong type)
146            tags_prop = properties.get("Tags")
147            if not tags_prop:
148                logger.warning("Database does not have a 'Tags' property (optional)")
149            elif tags_prop.get("type") != "multi_select":  # noqa: PLR2004
150                logger.warning(
151                    "'Tags' property should be of type 'multi_select', got '%(type)s'",
152                    {"type": tags_prop.get("type")},
153                )
154
155            logger.info(" Database properties validation passed")
156
157        except Exception as e:
158            logger.exception(
159                "Database property validation failed"
160                if isinstance(e, DatabasePropertyValidationError)  # noqa: TRY401
161                else "Failed to validate database properties"
162            )
163            raise

165    def get_existing_page_titles(self) -> set[str]:
166        """Get all existing page titles from the database.
167
168        Returns:
169            Set of page titles currently in the database
170        """
171        logger.info("Fetching existing pages from Notion database")
172        existing_titles: set[str] = set()
173
174        try:
175            # Step 1: Retrieve database to get data sources
176            database = self.client.databases.retrieve(database_id=self.database_id)
177            data_sources = database.get("data_sources", [])  # ty:ignore[unresolved-attribute]
178
179            if not data_sources:
180                logger.warning("No data sources found in database")
181                return existing_titles
182
183            # Step 2: Query each data source for pages
184            for data_source in data_sources:
185                data_source_id = data_source.get("id")
186                if not data_source_id:
187                    continue
188
189                logger.debug("Querying data source: %(id)s", {"id": data_source_id})
190
191                # Query data source with pagination
192                has_more = True
193                start_cursor = None
194
195                while has_more:
196                    # Build request body for data_sources query
197                    body: dict[str, Any] = {"page_size": 100}
198                    if start_cursor:
199                        body["start_cursor"] = start_cursor
200
201                    # Execute query using data_sources API
202                    response_dict = self.client.request(
203                        method="POST",
204                        path=f"data_sources/{data_source_id}/query",
205                        body=body,
206                    )
207                    response = QueryDatabaseResponse.model_validate(response_dict)
208
209                    for page in response.results:
210                        # Extract title from properties
211                        properties = page.get("properties", {})
212                        title_prop = properties.get("Title") or properties.get("Name")
213                        if title_prop and title_prop.get("type") == "title":  # noqa: PLR2004
214                            title_content = title_prop.get("title", [])
215                            if title_content:
216                                title_text = title_content[0].get("plain_text", "")
217                                if title_text:
218                                    existing_titles.add(title_text)
219
220                    has_more = response.has_more
221                    start_cursor = response.next_cursor
222
223            logger.info("Found %(count)d existing pages in Notion", {"count": len(existing_titles)})
224        except Exception:
225            logger.exception("Failed to fetch existing pages from Notion")
226            raise
227        else:
228            return existing_titles

231        self,
232        title: str,
233        scrapbox_url: str,
234        created_date: datetime,
235        tags: list[str],
236    ) -> dict:
237        """Create a new page in the Notion database.
238
239        Args:
240            title: Page title
241            scrapbox_url: URL to the original Scrapbox page
242            created_date: Original creation date from Scrapbox
243            tags: List of tags
244
245        Returns:
246            Created page object from Notion API
247
248        Raises:
249            Exception: If page creation fails
250        """
251        logger.debug("Creating Notion page: %(title)s", {"title": title})
252
253        properties = {
254            "Title": {
255                "type": "title",
256                "title": [{"type": "text", "text": {"content": title}}],
257            },
258            "Scrapbox URL": {"type": "url", "url": scrapbox_url},
259            "Created Date": {
260                "type": "date",
261                "date": {"start": created_date.isoformat()},
262            },
263        }
264
265        if tags:
266            properties["Tags"] = {
267                "type": "multi_select",
268                "multi_select": [{"name": tag} for tag in tags],
269            }
270
271        try:
272            # Create request object
273            create_request = CreatePageRequest(
274                parent={"database_id": self.database_id},
275                properties=properties,
276            )
277
278            # Execute request
279            response: dict | SyncAsync = self.client.pages.create(
280                **create_request.model_dump(mode="json", exclude_none=True)
281            )
282
283            # Return raw dict instead of validating with Page model
284            # because pydantic-api-models-notion may not support all parent types (e.g., data_source_id)
285            # Type assertion: client is used synchronously, so response is a dict
286            response_dict = cast("dict", response)
287            logger.info("Created page: %(title)s (ID: %(id)s)", {"title": title, "id": response_dict["id"]})
288        except Exception:
289            logger.exception("Failed to create page: %(title)s", {"title": title})
290            raise
291        else:
292            return response_dict

328    def append_blocks(self, page_id: UUID, blocks: list[BlockObject]) -> None:
329        """Append blocks to a Notion page.
330
331        Args:
332            page_id: Notion page ID
333            blocks: List of block objects to append (can include table blocks with children)
334
335        Raises:
336            Exception: If block append fails
337        """
338        if not blocks:
339            logger.debug("No blocks to append for page: %(page_id)s", {"page_id": page_id})
340            return
341
342        logger.debug("Appending %(count)d blocks to page: %(page_id)s", {"count": len(blocks), "page_id": page_id})
343
344        try:
345            # Notion API has a limit of 100 blocks per request
346            batch_size = 100
347            for i in range(0, len(blocks), batch_size):
348                batch = blocks[i : i + batch_size]
349                # Convert pydantic models to dicts, or use dict directly
350                batch_dicts = []
351                for block in batch:
352                    # Handle table blocks with children
353                    if isinstance(block, TableBlockWithChildren):
354                        # Convert table block and its children
355                        table_dict = block.block.model_dump(mode="json", exclude_none=True)
356                        # Add children to the table object, not at the top level
357                        table_dict["table"]["children"] = [
358                            child.model_dump(mode="json", exclude_none=True) for child in block.children
359                        ]
360                        batch_dicts.append(table_dict)
361                    elif hasattr(block, "model_dump"):
362                        batch_dicts.append(block.model_dump(mode="json", exclude_none=True))
363                    else:
364                        batch_dicts.append(block)
365
366                self.client.blocks.children.append(block_id=str(page_id), children=batch_dicts)
367                logger.debug(
368                    "Appended batch %(batch_num)d (%(count)d blocks)",
369                    {"batch_num": i // batch_size + 1, "count": len(batch)},
370                )
371
372            logger.info("Successfully appended %(count)d blocks", {"count": len(blocks)})
373        except Exception:
374            logger.exception("Failed to append blocks to page: %(page_id)s", {"page_id": page_id})
375            raise

445    def create_code_block(self, code: str, language: str = "plain text") -> CodeBlock | list[CodeBlock]:
446        """Create a code block.
447
448        Args:
449            code: Code content
450            language: Programming language (default: "plain text")
451
452        Returns:
453            Code block object, or list of code blocks if content exceeds 2000 characters
454        """
455        # Notion API limit: code.rich_text[0].text.content.length should be  2000
456        # Note: Notion API uses JavaScript's string.length which counts UTF-16 code units
457        MAX_CODE_LENGTH = 2000  # noqa: N806
458
459        def utf16_length(s: str) -> int:
460            """Calculate string length as UTF-16 code units (JavaScript string.length equivalent)."""
461            return len(s.encode("utf-16-le")) // 2
462
463        def split_at_utf16_boundary(s: str, max_utf16_length: int) -> str:
464            """Split string at UTF-16 boundary, ensuring we don't exceed max_utf16_length."""
465            # Binary search for the right split point
466            left, right = 0, len(s)
467            while left < right:
468                mid = (left + right + 1) // 2
469                if utf16_length(s[:mid]) <= max_utf16_length:
470                    left = mid
471                else:
472                    right = mid - 1
473            return s[:left]
474
475        if utf16_length(code) <= MAX_CODE_LENGTH:
476            return CodeBlock.new(code=code, language=language)
477
478        # Split code into chunks based on UTF-16 code units
479        num_blocks = (utf16_length(code) + MAX_CODE_LENGTH - 1) // MAX_CODE_LENGTH
480        logger.warning(
481            "Code block exceeds %(max)d characters (%(length)d UTF-16 code units, language=%(lang)s). "
482            "Splitting into %(num_blocks)d blocks.",
483            {"max": MAX_CODE_LENGTH, "length": utf16_length(code), "lang": language, "num_blocks": num_blocks},
484        )
485
486        blocks = []
487        remaining = code
488        while remaining:
489            # Find the split point that doesn't exceed MAX_CODE_LENGTH UTF-16 code units
490            chunk = split_at_utf16_boundary(remaining, MAX_CODE_LENGTH)
491            if not chunk:  # Safety check
492                # If we can't fit even one character, something is very wrong
493                logger.error("Cannot split code block - single character exceeds limit")
494                break
495
496            blocks.append(CodeBlock.new(code=chunk, language=language))
497            remaining = remaining[len(chunk) :]
498
499            logger.debug(
500                "Created code block chunk %(current)d/%(total)d (%(size)d UTF-16 code units)",
501                {"current": len(blocks), "total": num_blocks, "size": utf16_length(chunk)},
502            )
503
504        return blocks

592    def create_table_block(
593        self, table_rows: list[list[str]], *, has_column_header: bool = True, has_row_header: bool = False
594    ) -> TableBlockWithChildren | list[TableBlockWithChildren]:
595        """Create a table block with rows.
596
597        Args:
598            table_rows: List of rows, each row is a list of cell contents
599            has_column_header: Whether the first row is a header
600            has_row_header: Whether the first column is a header
601
602        Returns:
603            TableBlockWithChildren instance, or list of instances if table exceeds 100 rows
604        """
605        # Notion API limit: table.children.length should be  100
606        MAX_TABLE_ROWS = 100  # noqa: N806
607
608        if not table_rows:
609            msg = "Table must have at least one row"
610            raise ValueError(msg)
611
612        # Determine table width from the maximum row length
613        table_width = max(len(row) for row in table_rows)
614
615        # Normalize all rows to have the same width (pad with empty strings)
616        normalized_rows = [
617            row + [""] * (table_width - len(row)) if len(row) < table_width else row for row in table_rows
618        ]
619
620        # Check if table exceeds max rows limit
621        if len(normalized_rows) > MAX_TABLE_ROWS:
622            logger.warning(
623                "Table has %(count)d rows (max %(max)d). Splitting into multiple tables.",
624                {"count": len(normalized_rows), "max": MAX_TABLE_ROWS},
625            )
626
627            # Split table into chunks
628            result_tables = []
629            header_row = normalized_rows[0] if has_column_header else None
630
631            # Calculate effective max rows per table (including header if present)
632            effective_max = MAX_TABLE_ROWS - 1 if has_column_header else MAX_TABLE_ROWS
633
634            # Start from index 1 if there's a header, otherwise from 0
635            start_idx = 1 if has_column_header else 0
636
637            for i in range(start_idx, len(normalized_rows), effective_max):
638                chunk_rows = normalized_rows[i : i + effective_max]
639
640                # Add header to each chunk if has_column_header

740    def _sanitize_url(url: str) -> str | None:
741        """Sanitize and validate URL for Notion API.
742
743        Args:
744            url: URL to sanitize
745
746        Returns:
747            Sanitized URL or None if invalid
748        """
749        if not url:
750            return None
751
752        try:
753            # Remove trailing characters that are likely from text extraction errors
754            # These patterns often appear when URLs are extracted from code or markdown
755            original_url = url
756            url = url.rstrip("'\">`)")
757
758            # Parse the URL
759            parsed = urllib.parse.urlparse(url)
760
761            # Check if scheme is present and valid
762            if not parsed.scheme or parsed.scheme not in ("http", "https"):
763                logger.debug(
764                    "Invalid URL scheme: %(url)s (scheme: %(scheme)s)", {"url": original_url, "scheme": parsed.scheme}
765                )
766                return None
767
768            # Check if netloc (domain) is present
769            if not parsed.netloc:
770                logger.debug("Invalid URL, missing netloc: %(url)s", {"url": original_url})
771                return None
772
773            # Skip localhost URLs - treat them as plain text
774            if parsed.netloc in ("localhost", "127.0.0.1", "0.0.0.0") or parsed.netloc.startswith(  # noqa: S104
775                ("localhost:", "127.0.0.1:", "0.0.0.0:")
776            ):
777                logger.debug("Skipping localhost URL (treating as plain text): %(url)s", {"url": url})
778                return None
779
780            # Decode then re-encode to avoid double-encoding
781            # unquote handles %XX encoded characters
782            decoded_path = urllib.parse.unquote(parsed.path)
783            decoded_query = urllib.parse.unquote(parsed.query)
784            decoded_fragment = urllib.parse.unquote(parsed.fragment)
785
786            # Remove trailing slash from .git/ paths (common in git URLs)
787            if decoded_path.endswith(".git/"):
788                decoded_path = decoded_path[:-1]
789
790            # Reconstruct URL with proper encoding
791            # This ensures that fragments and query parameters are properly encoded
792            # For fragments, preserve common characters used in URL fragments:
793            # : (colon), ~ (tilde), = (equals), ; (semicolon), - (hyphen)
794            sanitized = urllib.parse.urlunparse(
795                (
796                    parsed.scheme,
797                    parsed.netloc,
798                    urllib.parse.quote(decoded_path, safe="/"),
799                    parsed.params,
800                    urllib.parse.quote(decoded_query, safe="=&:"),
801                    urllib.parse.quote(decoded_fragment, safe=":~=;-"),
802                )
803            )
804
805            if sanitized != original_url:
806                logger.debug(
807                    "Sanitized URL: %(original)s -> %(sanitized)s", {"original": original_url, "sanitized": sanitized}
808                )
809        except Exception:
810            logger.exception("Failed to sanitize URL: %(url)s", {"url": url})
811            return None
812        return sanitized

447        """Create a code block.
448
449        Args:
450            code: Code content
451            language: Programming language (default: "plain text")
452
453        Returns:
454            Code block object, or list of code blocks if content exceeds 2000 characters
455        """
456        # Notion API limit: code.rich_text[0].text.content.length should be  2000
457        # Note: Notion API uses JavaScript's string.length which counts UTF-16 code units
458        MAX_CODE_LENGTH = 2000  # noqa: N806
459
460        def utf16_length(s: str) -> int:
461            """Calculate string length as UTF-16 code units (JavaScript string.length equivalent)."""
462            return len(s.encode("utf-16-le")) // 2
463
464        def split_at_utf16_boundary(s: str, max_utf16_length: int) -> str:
465            """Split string at UTF-16 boundary, ensuring we don't exceed max_utf16_length."""
466            # Binary search for the right split point
467            left, right = 0, len(s)
468            while left < right:
469                mid = (left + right + 1) // 2
470                if utf16_length(s[:mid]) <= max_utf16_length:
471                    left = mid
472                else:
473                    right = mid - 1
474            return s[:left]
475
476        if utf16_length(code) <= MAX_CODE_LENGTH:
477            return CodeBlock.new(code=code, language=language)
478
479        # Split code into chunks based on UTF-16 code units
480        num_blocks = (utf16_length(code) + MAX_CODE_LENGTH - 1) // MAX_CODE_LENGTH
481        logger.warning(
482            "Code block exceeds %(max)d characters (%(length)d UTF-16 code units, language=%(lang)s). "
483            "Splitting into %(num_blocks)d blocks.",
484            {"max": MAX_CODE_LENGTH, "length": utf16_length(code), "lang": language, "num_blocks": num_blocks},
485        )
486
487        blocks = []
488        remaining = code
489        while remaining:
490            # Find the split point that doesn't exceed MAX_CODE_LENGTH UTF-16 code units
491            chunk = split_at_utf16_boundary(remaining, MAX_CODE_LENGTH)
492            if not chunk:  # Safety check
493                # If we can't fit even one character, something is very wrong
494                logger.error("Cannot split code block - single character exceeds limit")
495                break
496
497            blocks.append(CodeBlock.new(code=chunk, language=language))
498            remaining = remaining[len(chunk) :]
499
500            logger.debug(
501                "Created code block chunk %(current)d/%(total)d (%(size)d UTF-16 code units)",
502                {"current": len(blocks), "total": num_blocks, "size": utf16_length(chunk)},
503            )
504
505        return blocks

608
609        if not table_rows:
610            msg = "Table must have at least one row"
611            raise ValueError(msg)
612
613        # Determine table width from the maximum row length
614        table_width = max(len(row) for row in table_rows)
615
616        # Normalize all rows to have the same width (pad with empty strings)
617        normalized_rows = [
618            row + [""] * (table_width - len(row)) if len(row) < table_width else row for row in table_rows

626            )
627
628            # Split table into chunks
629            result_tables = []

631

633            effective_max = MAX_TABLE_ROWS - 1 if has_column_header else MAX_TABLE_ROWS
634
635            # Start from index 1 if there's a header, otherwise from 0
636            start_idx = 1 if has_column_header else 0
637
638            for i in range(start_idx, len(normalized_rows), effective_max):
639                chunk_rows = normalized_rows[i : i + effective_max]

641                # Add header to each chunk if has_column_header

662
663        # Create table row blocks

741    def _sanitize_url(url: str) -> str | None:
742        """Sanitize and validate URL for Notion API.
743
744        Args:
745            url: URL to sanitize
746
747        Returns:
748            Sanitized URL or None if invalid
749        """
750        if not url:
751            return None
752
753        try:
754            # Remove trailing characters that are likely from text extraction errors
755            # These patterns often appear when URLs are extracted from code or markdown
756            original_url = url
757            url = url.rstrip("'\">`)")
758
759            # Parse the URL
760            parsed = urllib.parse.urlparse(url)
761
762            # Check if scheme is present and valid
763            if not parsed.scheme or parsed.scheme not in ("http", "https"):
764                logger.debug(
765                    "Invalid URL scheme: %(url)s (scheme: %(scheme)s)", {"url": original_url, "scheme": parsed.scheme}
766                )
767                return None
768
769            # Check if netloc (domain) is present
770            if not parsed.netloc:
771                logger.debug("Invalid URL, missing netloc: %(url)s", {"url": original_url})
772                return None
773
774            # Skip localhost URLs - treat them as plain text
775            if parsed.netloc in ("localhost", "127.0.0.1", "0.0.0.0") or parsed.netloc.startswith(  # noqa: S104
776                ("localhost:", "127.0.0.1:", "0.0.0.0:")
777            ):
778                logger.debug("Skipping localhost URL (treating as plain text): %(url)s", {"url": url})
779                return None
780
781            # Decode then re-encode to avoid double-encoding
782            # unquote handles %XX encoded characters
783            decoded_path = urllib.parse.unquote(parsed.path)
784            decoded_query = urllib.parse.unquote(parsed.query)
785            decoded_fragment = urllib.parse.unquote(parsed.fragment)
786
787            # Remove trailing slash from .git/ paths (common in git URLs)
788            if decoded_path.endswith(".git/"):
789                decoded_path = decoded_path[:-1]
790
791            # Reconstruct URL with proper encoding
792            # This ensures that fragments and query parameters are properly encoded
793            # For fragments, preserve common characters used in URL fragments:
794            # : (colon), ~ (tilde), = (equals), ; (semicolon), - (hyphen)
795            sanitized = urllib.parse.urlunparse(
796                (
797                    parsed.scheme,
798                    parsed.netloc,
799                    urllib.parse.quote(decoded_path, safe="/"),
800                    parsed.params,
801                    urllib.parse.quote(decoded_query, safe="=&:"),
802                    urllib.parse.quote(decoded_fragment, safe=":~=;-"),
803                )
804            )
805
806            if sanitized != original_url:
807                logger.debug(
808                    "Sanitized URL: %(original)s -> %(sanitized)s", {"original": original_url, "sanitized": sanitized}
809                )
810        except Exception:
811            logger.exception("Failed to sanitize URL: %(url)s", {"url": url})
812            return None
813        return sanitized
